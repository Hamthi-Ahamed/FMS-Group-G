
CREATE TABLE users (
    id         INT          PRIMARY KEY AUTO_INCREMENT,
    username   VARCHAR(100) NOT NULL UNIQUE,
    password   VARCHAR(255) NOT NULL,           -- store hashed value
    role       ENUM('Admin','Student','Lecturer') NOT NULL,
    created_at TIMESTAMP    DEFAULT CURRENT_TIMESTAMP
);


CREATE TABLE departments (
    id             INT          PRIMARY KEY AUTO_INCREMENT,
    name           VARCHAR(100) NOT NULL UNIQUE,
    hod_lecturer_id INT          DEFAULT NULL,   -- FK added after LECTURERS
    no_of_staff    INT          DEFAULT 0
);


CREATE TABLE degrees (
    id             INT          PRIMARY KEY AUTO_INCREMENT,
    name           VARCHAR(100) NOT NULL UNIQUE,
    department_id  INT          NOT NULL,
    no_of_students INT          DEFAULT 0,
    CONSTRAINT fk_degree_dept
        FOREIGN KEY (department_id) REFERENCES departments(id)
        ON DELETE RESTRICT ON UPDATE CASCADE
);


CREATE TABLE lecturers (
    id            INT          PRIMARY KEY AUTO_INCREMENT,
    user_id       INT          NOT NULL UNIQUE,
    full_name     VARCHAR(100) NOT NULL,
    email         VARCHAR(150) NOT NULL UNIQUE,
    mobile_number VARCHAR(20),
    department_id INT          NOT NULL,
    CONSTRAINT fk_lecturer_user
        FOREIGN KEY (user_id)       REFERENCES users(id)
        ON DELETE CASCADE  ON UPDATE CASCADE,
    CONSTRAINT fk_lecturer_dept
        FOREIGN KEY (department_id) REFERENCES departments(id)
        ON DELETE RESTRICT ON UPDATE CASCADE
);


ALTER TABLE departments
    ADD CONSTRAINT fk_dept_hod
        FOREIGN KEY (hod_lecturer_id) REFERENCES lecturers(id)
        ON DELETE SET NULL ON UPDATE CASCADE;


CREATE TABLE students (
    id            INT          PRIMARY KEY AUTO_INCREMENT,
    user_id       INT          NOT NULL UNIQUE,
    student_id    VARCHAR(30)  NOT NULL UNIQUE,  -- e.g. ET/2022/011
    full_name     VARCHAR(100) NOT NULL,
    email         VARCHAR(150) NOT NULL UNIQUE,
    mobile_number VARCHAR(20),
    degree_id     INT          NOT NULL,
    CONSTRAINT fk_student_user
        FOREIGN KEY (user_id)   REFERENCES users(id)
        ON DELETE CASCADE  ON UPDATE CASCADE,
    CONSTRAINT fk_student_degree
        FOREIGN KEY (degree_id) REFERENCES degrees(id)
        ON DELETE RESTRICT ON UPDATE CASCADE
);


CREATE TABLE courses (
    id          INT          PRIMARY KEY AUTO_INCREMENT,
    course_code VARCHAR(20)  NOT NULL UNIQUE,    -- e.g. ETEC 21062
    course_name VARCHAR(150) NOT NULL,
    credits     INT          NOT NULL DEFAULT 2,
    lecturer_id INT          NOT NULL,
    CONSTRAINT fk_course_lecturer
        FOREIGN KEY (lecturer_id) REFERENCES lecturers(id)
        ON DELETE RESTRICT ON UPDATE CASCADE
);


CREATE TABLE enrollments (
    id         INT         PRIMARY KEY AUTO_INCREMENT,
    student_id INT         NOT NULL,
    course_id  INT         NOT NULL,
    grade      VARCHAR(5)  DEFAULT NULL,         -- e.g. A+, B, C
    enrolled_at TIMESTAMP  DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_enroll_student
        FOREIGN KEY (student_id) REFERENCES students(id)
        ON DELETE CASCADE  ON UPDATE CASCADE,
    CONSTRAINT fk_enroll_course
        FOREIGN KEY (course_id)  REFERENCES courses(id)
        ON DELETE CASCADE  ON UPDATE CASCADE,
    CONSTRAINT uq_enrollment UNIQUE (student_id, course_id)
);


CREATE TABLE timetable (
    id        INT  PRIMARY KEY AUTO_INCREMENT,
    course_id INT  NOT NULL,
    day       ENUM('Monday','Tuesday','Wednesday','Thursday','Friday') NOT NULL,
    time_slot VARCHAR(10) NOT NULL,              -- e.g. 08:00, 10:00
    CONSTRAINT fk_timetable_course
        FOREIGN KEY (course_id) REFERENCES courses(id)
        ON DELETE CASCADE ON UPDATE CASCADE,
    CONSTRAINT uq_timetable UNIQUE (day, time_slot, course_id)
);
